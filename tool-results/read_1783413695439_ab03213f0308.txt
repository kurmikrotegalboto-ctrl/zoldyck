    91тЖТ    91тЖТfunction mergeSnapshots(
    92тЖТ    92тЖТ  existing: SnapshotData[],
    93тЖТ    93тЖТ  newSnapshot: SnapshotData
    94тЖТ    94тЖТ): SnapshotData[] {
    95тЖТ    95тЖТ  const arr = [...existing];
    96тЖТ    96тЖТ  const idx = arr.findIndex((s) => s.dateSort === newSnapshot.dateSort);
    97тЖТ    97тЖТ  if (idx >= 0) {
    98тЖТ    98тЖТ    const updatedUnits = [...arr[idx].units];
    99тЖТ    99тЖТ    for (const unit of newSnapshot.units) {
   100тЖТ   100тЖТ      const uIdx = updatedUnits.findIndex((u) => u.code === unit.code);
   101тЖТ   101тЖТ      if (uIdx >= 0) updatedUnits[uIdx] = unit;
   102тЖТ   102тЖТ      else updatedUnits.push(unit);
   103тЖТ   103тЖТ    }
   104тЖТ   104тЖТ    arr[idx] = { ...arr[idx], units: updatedUnits };
   105тЖТ   105тЖТ  } else {
   106тЖТ   106тЖТ    arr.push(newSnapshot);
   107тЖТ   107тЖТ  }
   108тЖТ   108тЖТ  arr.sort((a, b) => a.dateSort.localeCompare(b.dateSort));
   109тЖТ   109тЖТ  return arr;
   110тЖТ   110тЖТ}
