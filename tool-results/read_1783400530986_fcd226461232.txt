    90тЖТ    90тЖТfunction mergeSnapshots(
    91тЖТ    91тЖТ  existing: SnapshotData[],
    92тЖТ    92тЖТ  newSnapshot: SnapshotData
    93тЖТ    93тЖТ): SnapshotData[] {
    94тЖТ    94тЖТ  const arr = [...existing];
    95тЖТ    95тЖТ  const idx = arr.findIndex((s) => s.dateSort === newSnapshot.dateSort);
    96тЖТ    96тЖТ  if (idx >= 0) {
    97тЖТ    97тЖТ    const updatedUnits = [...arr[idx].units];
    98тЖТ    98тЖТ    for (const unit of newSnapshot.units) {
    99тЖТ    99тЖТ      const uIdx = updatedUnits.findIndex((u) => u.code === unit.code);
   100тЖТ   100тЖТ      if (uIdx >= 0) updatedUnits[uIdx] = unit;
   101тЖТ   101тЖТ      else updatedUnits.push(unit);
   102тЖТ   102тЖТ    }
   103тЖТ   103тЖТ    arr[idx] = { ...arr[idx], units: updatedUnits };
   104тЖТ   104тЖТ  } else {
   105тЖТ   105тЖТ    arr.push(newSnapshot);
   106тЖТ   106тЖТ  }
   107тЖТ   107тЖТ  arr.sort((a, b) => a.dateSort.localeCompare(b.dateSort));
   108тЖТ   108тЖТ  return arr;
   109тЖТ   109тЖТ}
